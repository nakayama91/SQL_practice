select *
// *ですべてのカラムを呼び出す。nameやpriceなど指定可能
from データベース名（user,purchasesなど）
order by price desc //値段を降順に並べる。ascで昇順
limit 5; //最大5件を呼び出す。limitはクエリの最後に記述

where category = "食費"
AND　character_name = "ひつじ"
//食費とひつじを含むデータを呼び出す

where price is null　//priceが空欄のデータを呼び出す
where price is not null //priceが空欄でないデータを呼び出す
where not character_name = "ひつじ"　//character_nameがひつじでないデータを呼び出す
where purchased_at <= "2017-08-01" //日付以前のデータを呼び出す
where name LIKE "%プリン%"　//プリンを含むデータを呼び出す

//グループ化、
select sum(price), character_name, purchases_at
from purchases
where category = '雑費'
group by character_name, purchases_at
